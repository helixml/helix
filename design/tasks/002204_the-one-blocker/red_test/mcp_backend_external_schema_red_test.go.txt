package server

// RED TDD proof for helix-specs task 002204_the-one-blocker.
//
// It demonstrates that the external MCP proxy's current tool-schema
// reconstruction (mcp_backend_external.go ~L322-358) flattens EVERY parameter
// to type:"string", collapsing array params like create_bot.tools/topics and
// subscribe.topicIds — the exact cause of the Bot's
// "cannot unmarshal string into []string" blocker.
//
// TestProxyReadvertise_Current REPLICATES the production construction verbatim
// and asserts the array survives → it FAILS (RED), pinning the defect.
// TestProxyReadvertise_Fixed shows the proposed NewToolWithRawSchema
// passthrough preserves the array → it PASSES, validating the fix.

import (
	"encoding/json"
	"testing"

	"github.com/mark3labs/mcp-go/mcp"
)

// upstreamCreateBotTool mimics what externalClient.ListTools() returns from
// the org MCP server: create_bot with `tools` as a proper array-of-string
// (the org server is verified correct by TestSchemaWireArrayParams).
func upstreamCreateBotTool() mcp.Tool {
	return mcp.Tool{
		Name:        "create_bot",
		Description: "Create a new Bot in one call.",
		InputSchema: mcp.ToolInputSchema{
			Type: "object",
			Properties: map[string]any{
				"content": map[string]any{
					"type":        "string",
					"description": "The bot's prompt.",
				},
				"tools": map[string]any{
					"type":        "array",
					"description": "MCP tools to grant.",
					"items":       map[string]any{"type": "string"},
				},
			},
			Required: []string{"content", "tools"},
		},
	}
}

// paramType marshals a mark3labs tool exactly as it is served over the wire and
// returns properties.<name>.type as seen by the client (Zed/Qwen).
func paramType(t *testing.T, tool mcp.Tool, param string) string {
	t.Helper()
	raw, err := json.Marshal(tool)
	if err != nil {
		t.Fatalf("marshal tool: %v", err)
	}
	var parsed struct {
		InputSchema struct {
			Properties map[string]struct {
				Type json.RawMessage `json:"type"`
			} `json:"properties"`
		} `json:"inputSchema"`
	}
	if err := json.Unmarshal(raw, &parsed); err != nil {
		t.Fatalf("unmarshal served tool: %v", err)
	}
	var typ string
	_ = json.Unmarshal(parsed.InputSchema.Properties[param].Type, &typ)
	return typ
}

// TestProxyReadvertise_Current replicates mcp_backend_external.go L322-358.
// EXPECTED TO FAIL until the proxy stops flattening params to string.
func TestProxyReadvertise_Current(t *testing.T) {
	upstream := upstreamCreateBotTool()

	// --- verbatim copy of production reconstruction ---
	var opts []mcp.ToolOption
	opts = append(opts, mcp.WithDescription(upstream.Description))
	if upstream.InputSchema.Properties != nil {
		required := make(map[string]bool)
		for _, r := range upstream.InputSchema.Required {
			required[r] = true
		}
		for propName, propDef := range upstream.InputSchema.Properties {
			propMap, ok := propDef.(map[string]any)
			if !ok {
				continue
			}
			desc := ""
			if d, ok := propMap["description"].(string); ok {
				desc = d
			}
			if required[propName] {
				opts = append(opts, mcp.WithString(propName, mcp.Required(), mcp.Description(desc)))
			} else {
				opts = append(opts, mcp.WithString(propName, mcp.Description(desc)))
			}
		}
	}
	served := mcp.NewTool(upstream.Name, opts...)
	// --- end copy ---

	if got := paramType(t, served, "tools"); got != "array" {
		t.Fatalf("create_bot.tools re-advertised as %q, want \"array\" — proxy flattened the array to string (the reported blocker)", got)
	}
}

// TestProxyReadvertise_Fixed proves the passthrough fix preserves the array.
func TestProxyReadvertise_Fixed(t *testing.T) {
	upstream := upstreamCreateBotTool()

	raw, err := json.Marshal(upstream.InputSchema)
	if err != nil {
		t.Fatalf("marshal upstream schema: %v", err)
	}
	served := mcp.NewToolWithRawSchema(upstream.Name, upstream.Description, raw)

	if got := paramType(t, served, "tools"); got != "array" {
		t.Fatalf("create_bot.tools = %q, want \"array\" after passthrough fix", got)
	}
	if got := paramType(t, served, "content"); got != "string" {
		t.Fatalf("create_bot.content = %q, want \"string\" (non-array must stay intact)", got)
	}
}
